(function() {
  function updateClock() {
    const now = new Date();
    const hours = String(now.getHours()).padStart(2, '0');
    const minutes = String(now.getMinutes()).padStart(2, '0');
    const seconds = String(now.getSeconds()).padStart(2, '0');
    const timeString = `${hours}:${minutes}:${seconds}`;
    
    const clockSvg = document.getElementById('clockSvg');
    if (clockSvg) {
      try {
        const svgDoc = clockSvg.contentDocument || clockSvg.getSVGDocument();
        if (svgDoc) {
          const digitalTime = svgDoc.getElementById('digitalTime');
          const dateDisplay = svgDoc.getElementById('dateDisplay');
          if (digitalTime) digitalTime.textContent = timeString;
          
          const monthNames = ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"];
          const dayNames = ["Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"];
          if (dateDisplay) {
            dateDisplay.textContent = `${dayNames[now.getDay()]}, ${monthNames[now.getMonth()]} ${now.getDate()}`;
          }
        }
      } catch(e) {
        // SVG updates handled by SVG animations
      }
    }
  }
  updateClock();
  setInterval(updateClock, 1000);
})();
</script>
